-- LIB_BOOKS 생성
CREATE TABLE LIB_BOOKS (BID VARCHAR2(26 BYTE), TITLE VARCHAR2(512 BYTE), AUTHOR VARCHAR2(1024 BYTE), COPYRIGHT VARCHAR2(128 BYTE), YEAR NUMBER(4,0), CONSTRAINT LIB_BOOKS_PK PRIMARY KEY (BID))

-- LIB_RENTAL
CREATE TABLE LIB_RENTAL (RENTAL_ID NUMBER(8,0), BID VARCHAR2(26 BYTE), U_ID NUMBER(8,0), DATE_START DATE NOT NULL, DATE_RETURN DATE, RENEW NUMBER(2,0), CONSTRAINT LIB_RENTAL_PK PRIMARY KEY (RENTAL_ID), CONSTRAINT LIB_RENTAL_FK1 FOREIGN KEY (BID) REFERENCES LIB_BOOKS (BID) ENABLE, CONSTRAINT LIB_RENTAL_FK2 FOREIGN KEY (U_ID) REFERENCES LIB_USERS (U_ID) ENABLE)

-- LIB_USERS
CREATE TABLE LIB_USERS (U_ID NUMBER(8,0), ID VARCHAR2(20 BYTE) UNIQUE NOT NULL, PW VARCHAR2(40 BYTE) NOT NULL, PHONENUMBER VARCHAR2(13 BYTE) NOT NULL, EMAIL VARCHAR2(100 BYTE) UNIQUE NOT NULL, CONSTRAINT LIB_USERS_PK PRIMARY KEY (U_ID))

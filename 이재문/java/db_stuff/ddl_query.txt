-- AI.LIB_BOOKS 생성
CREATE TABLE AI.LIB_BOOKS
(
	BID VARCHAR2(26 BYTE) NOT NULL ENABLE, 
	TITLE VARCHAR2(512 BYTE), 
	AUTHOR VARCHAR2(1024 BYTE), 
	COPYRIGHT VARCHAR2(128 BYTE), 
	YEAR NUMBER(4,0), 
	 CONSTRAINT "LIB_BOOKS_PK" PRIMARY KEY ("BID")
);

-- AI.LIB_RENTAL
CREATE TABLE AI.LIB_RENTAL 
(	REN_NUM NUMBER NOT NULL ENABLE, 
	REN_DATE DATE NOT NULL ENABLE, 
	RTN_DATE DATE, 
	PRE_DATE DATE, 
	SUN_ID VARCHAR2(20 BYTE), 
	REN_BID VARCHAR2(26 BYTE) NOT NULL ENABLE, 
	REN_USER VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	 CONSTRAINT "LIB_RENTAL_PK" PRIMARY KEY ("REN_NUM")
);

-- AI.LIB_USERS
CREATE TABLE AI.LIB_USERS 
(	LIB_UID VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	LIB_PASS VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	LIB_NAME VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	LIB_PHONE VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	 CONSTRAINT "LIB_USER_PK" PRIMARY KEY ("LIB_UID")
);
